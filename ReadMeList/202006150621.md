func searchSosu(max: Int) -> [Int] {
    let MAX = max
    var sosuArr = Array(repeating: 0, count: MAX+1)
    let sosuRange = Int(sqrt(Double(MAX)))
    sosuArr[0] = 1
    sosuArr[1] = 1
    for i in 2...sosuRange {
        if sosuArr[i] == 1 {
            continue
        }
        for n in stride(from: i+i, through: MAX, by: i) {
            if sosuArr[n] == 0 {
                sosuArr[n] = 1
            }
        }
    }
    return sosuArr
}

func solution2581() {
    var m = Int(readLine()!)!
    let n = Int(readLine()!)!
    
    let sosuArr = searchSosu(max: 10000)
    var sum = 0
    var min = n
    
    m = m >= 2 ? m : 2
    for i in stride(from: m, through: n, by: 1) {
        if sosuArr[i] == 0 {
            sum += i
            if min > i {
                min = i
            }
        }
    }
    sum = sum == 0 ? -1 : sum
    print(sum)
    if sum == -1 {
        return
    }
    print(min)
}
```

      * 소수찾을때 인덱스로접근 [100] 접근시 에러
      * +1개의 배열을 생성해줌으로써 예외처리

func solution2581() {
     var m = Int(readLine()!)!
     let n = Int(readLine()!)!

     var sum = 0
     var min = n
     m = m >= 2 ? m : 2
     
     for i in stride(from: m, through: n, by: 1) {
         var isSosu = true

         for n in 2..<i {
             if i % n == 0 {
                 isSosu = false
                 break
             }
         }
         if isSosu {
             sum += i
             if min > i {
                 min = i
             }
         }
     }
     sum = sum == 0 ? -1 : sum
     print(sum)
     if sum != -1 {
         print(min)
     }
     
 }
```

     let V = Int(input[2])!

             print(i)

         }
         result -= down
     }
 }

func soultion2869() {
