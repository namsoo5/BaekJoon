## 2869

``` swift
 func soultion2869() {
     let input = readLine()!.components(separatedBy: " ")
     let up = Int(input[0])!
     let down = Int(input[1])!
     let V = Int(input[2])!
     
     var result = 0
     for i in 1... {
         result += up
         
         if result >= V {
             print(i)
             return
         }
         result -= down
     }
 }
```

* 처음에 위와 같은 코드로 구현했으나 효율성에서 시간초과로 실패

* 시간이 중요한 문제는 규칙을 찾고 수식으로 푸는 방법으로 도전해보기

``` swift
func soultion2869() {
    let input = readLine()!.components(separatedBy: " ")
    let up = Int(input[0])!
    let down = Int(input[1])!
    let V = Int(input[2])!
    
    let gap = up - down
    let day = (V-up)/gap
    if (V-up)%gap == 0 {
        print(day+1)
    } else {
        print(day+2)
    }   
}
```
